apples=int(input("HELLO HARRY\nENTER NUMBER OF APPLES YOU GOT\n"))
mx=int(input("ENTER MAX NUMBER OF STUDENTS\n"))
mn=int(input("ENTER MIN NUMBER OF STUDENS\n"))
if mn==mx:
    print("THIS IS NOT A RANGE BECAUSE MAX NUMBERS AND MIN NUMBERS ARE EQUAL")
    exit()
for a in range(mn,mx+1):
    if apples%a==0:
        print(a,"IS DIVISOR OF",apples)
    else:
        print(a,"IS NOT DIVISOR OF",apples)
